      \ln a + \ln b & = & x \ln 4 + x \ln 5 \\
      \frac{\ln a}{\ln b } & = & \frac{\ln 4}{\ln 5} \\
      \ln \left(a - b \right)  & = & \frac{\ln 4}{\ln 5} \\

    \begin{eqnarray*}
